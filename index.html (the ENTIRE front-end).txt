  <script>
    const API = 'https://tavern-rota-backend.onrender.com'; // weâ€™ll update this later
    let user = null;
    function login(){
      const email = document.getElementById('email').value;
      const pw = document.getElementById('password').value;
      fetch(API+'/api/auth/login',{
        method:'POST',
        headers:{'Content-Type':'application/json'},
        body:JSON.stringify({email,password:pw})
      }).then(r=>r.json()).then(data=>{
        if(data.error){document.getElementById('msg').textContent=data.error;return;}
        user=data;
        document.getElementById('user').textContent=user.name;
        document.getElementById('role').textContent=user.role;
        document.getElementById('login').style.display='none';
        document.getElementById('rota').style.display='block';
        document.getElementById('genBtn').style.display=(user.role==='admin')?'inline-block':'none';
      });
    }
    function logout(){ location.reload() }
    function loadRota(){
      const m = document.getElementById('month').value;
      fetch(API+'/api/rota/'+m).then(r=>r.json()).then(data=>{
        const tb = document.getElementById('grid');
        tb.innerHTML='';
        for(let d=1;d<=31;d++){
          const tr = document.createElement('tr');
          const td1 = document.createElement('td'); td1.textContent=d;
          const td2 = document.createElement('td');
          if(!data[d]){ td2.textContent='UNSTAFFED'; tr.classList.add('red'); }
          else { td2.textContent='Staff #'+data[d]; }
          tr.append(td1,td2); tb.appendChild(tr);
        }
      });
    }
    function generateRota(){
      const m = document.getElementById('month').value;
      fetch(API+'/api/rota/'+m+'/generate',{method:'POST'})
        .then(r=>r.json()).then(_=>loadRota());
    }
  </script>
